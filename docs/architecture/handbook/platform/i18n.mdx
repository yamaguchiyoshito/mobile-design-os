## 2. 初期化

```ts
// src/lib/i18n.ts
import i18n              from 'i18next';
import { initReactI18next } from 'react-i18next';
import * as Localization from 'expo-localization';

// ─── 翻訳リソース ─────────────────────────────────────────────────────────
import ja from '@/locales/ja';
import en from '@/locales/en';

const SUPPORTED_LOCALES = ['ja', 'en'] as const;
export type SupportedLocale = (typeof SUPPORTED_LOCALES)[number];

// ─── 初期化 ──────────────────────────────────────────────────────────────
export function initializeI18n() {
  // 端末のロケールを取得し、サポート言語に丸める
  const deviceLocale  = Localization.getLocales()[0]?.languageCode ?? 'ja';
  const matchedLocale = SUPPORTED_LOCALES.find((l) => deviceLocale.startsWith(l)) ?? 'ja';

  i18n
    .use(initReactI18next)
    .init({
      lng:      matchedLocale,
      fallbackLng: 'ja',
      resources: {
        ja: { translation: ja },
        en: { translation: en },
      },
      interpolation: {
        escapeValue: false,  // React は XSS を自動エスケープするため不要
      },
      returnNull:    false,
      returnEmptyString: false,
    });
}

// ─── ロケール切替（アプリ内設定変更）────────────────────────────────────
export async function changeLocale(locale: SupportedLocale) {
  await i18n.changeLanguage(locale);
  await SecureStore.setItemAsync('app_locale', locale);
}

// ─── アプリ設定から保存済みロケールを復元 ────────────────────────────────
export async function restoreLocale() {
  const saved = await SecureStore.getItemAsync('app_locale') as SupportedLocale | null;
  if (saved && SUPPORTED_LOCALES.includes(saved)) {
    await i18n.changeLanguage(saved);
  }
}
```

```tsx
// app/_layout.tsx
import { initializeI18n, restoreLocale } from '@/lib/i18n';
initializeI18n();

export default function RootLayout() {
  useEffect(() => { restoreLocale(); }, []);
  // ...
}
```

```ts
  },

```

### キー命名規約

```
[ドメイン].[操作/状態][結果?]

例:
  order.submitSuccess     → 注文送信成功
  order.status.pending    → 注文ステータス: 処理中
  payment.error.declined  → 決済エラー: 拒否
  validation.required     → バリデーション: 必須
```

🔴 **MUST**: キーは全てカテゴリを持つ（ルートキー直書き禁止）  
🟡 **SHOULD**: キーファイルは `src/locales/[lang]/[domain].ts` にドメインごとに分割し、`index.ts` で統合する  
🔴 **MUST**: コンポーネント内に文言をハードコードしない（09_Message_Display と同じ規約）

## 4. `useTranslation` の使い方

```tsx
// ✅ 基本使用
const { t } = useTranslation();
<Text>{t('order.submitSuccess')}</Text>

// ✅ 変数展開
<Text>{t('validation.minLength', { min: 8 })}</Text>
// → "8文字以上入力してください"

// ✅ 複数形
<Text>{t('items.count', { count: 3 })}</Text>
// → "3件"

// ✅ コンポーネント（Trans を使う）
import { Trans } from 'react-i18next';
// locales/ja: "利用規約 <link>こちら</link> を確認してください"
<Trans
  i18nKey="auth.termsMessage"
  components={{
    link: <Pressable onPress={openTerms} accessibilityRole="link" />,
  }}
/>
```

🔴 **MUST NOT**: `t()` の戻り値を `dangerouslySetInnerHTML` / `innerHTML` に渡さない（RN では不要だが習慣として）

## 5. 日付・時刻フォーマット

`Intl.DateTimeFormat` を使う。`moment` / `date-fns` のフォーマット関数は使わない
（ロケール対応が自動で行われるため）。

```ts
// src/lib/formatDate.ts
import i18n from 'i18next';

// ─── 日付フォーマット ─────────────────────────────────────────────────────
export function formatDate(
  date: Date | string,
  style: 'short' | 'medium' | 'long' | 'full' = 'medium',
): string {
  const d      = typeof date === 'string' ? new Date(date) : date;
  const locale = i18n.language;

  return new Intl.DateTimeFormat(locale, { dateStyle: style }).format(d);
}

// ─── 時刻フォーマット ─────────────────────────────────────────────────────
export function formatTime(date: Date | string): string {
  const d      = typeof date === 'string' ? new Date(date) : date;
  const locale = i18n.language;

  return new Intl.DateTimeFormat(locale, { timeStyle: 'short' }).format(d);
}

// ─── 相対時刻（「3 分前」等）─────────────────────────────────────────────
export function formatRelativeTime(date: Date | string): string {
  const d         = typeof date === 'string' ? new Date(date) : date;
  const locale    = i18n.language;
  const formatter = new Intl.RelativeTimeFormat(locale, { numeric: 'auto' });
  const diffMs    = d.getTime() - Date.now();
  const diffSec   = Math.round(diffMs / 1000);
  const diffMin   = Math.round(diffSec / 60);
  const diffHour  = Math.round(diffMin / 60);
  const diffDay   = Math.round(diffHour / 24);

  if (Math.abs(diffSec) < 60)  return formatter.format(diffSec,  'second');
  if (Math.abs(diffMin) < 60)  return formatter.format(diffMin,  'minute');
  if (Math.abs(diffHour) < 24) return formatter.format(diffHour, 'hour');
  return formatter.format(diffDay, 'day');
}
```

```tsx
// 使用例
<Text>{formatDate('2026-03-15', 'long')}</Text>
// ja: "2026年3月15日"
// en: "March 15, 2026"

<Text>{formatRelativeTime('2026-03-15T09:00:00+09:00')}</Text>
// ja: "3 日前"
// en: "3 days ago"
```

## 6. 数値・通貨フォーマット

```ts
// src/lib/formatNumber.ts
import i18n from 'i18next';

// ─── 通貨フォーマット ─────────────────────────────────────────────────────
export function formatPrice(
  amount:   number,
  currency: string,
): string {
  const locale = i18n.language;
  return new Intl.NumberFormat(locale, {
    style:    'currency',
    currency,
    minimumFractionDigits: currency === 'JPY' ? 0 : 2,
    maximumFractionDigits: currency === 'JPY' ? 0 : 2,
  }).format(amount);
}

// ─── 数値フォーマット（カンマ区切り等）──────────────────────────────────
export function formatNumber(value: number, options?: Intl.NumberFormatOptions): string {
  return new Intl.NumberFormat(i18n.language, options).format(value);
}

// ─── 単位付きフォーマット ─────────────────────────────────────────────────
export function formatFileSize(bytes: number): string {
  if (bytes < 1024)       return `${bytes} B`;
  if (bytes < 1024 ** 2)  return `${(bytes / 1024).toFixed(1)} KB`;
  if (bytes < 1024 ** 3)  return `${(bytes / 1024 ** 2).toFixed(1)} MB`;
  return `${(bytes / 1024 ** 3).toFixed(2)} GB`;
}
```

## 7. RTL 対応

アラビア語・ヘブライ語等の RTL 言語を将来追加する場合の準備。

```tsx
// src/hooks/useLayoutDirection.ts
import * as Localization from 'expo-localization';
import { I18nManager }   from 'react-native';

export function useLayoutDirection() {
  const isRTL = Localization.getLocales()[0]?.textDirection === 'rtl';

  useEffect(() => {
    if (I18nManager.isRTL !== isRTL) {
      I18nManager.allowRTL(isRTL);
      I18nManager.forceRTL(isRTL);
      // RTL 変更後はアプリを再起動する必要がある
      // RNRestart.Restart(); // expo-updates の reloadAsync で代替可
    }
  }, [isRTL]);

  return { isRTL };
}
```

```tsx
// RTL 対応の NativeWind クラス
// ✅ 方向に依存しないクラスを使う
<View className="flex-row ms-4">   {/* ms = margin-start (RTL で右になる) */}
<View className="ps-3">           {/* ps = padding-start */}

// ❌ 絶対方向のクラスを使わない（RTL で崩れる）
<View className="flex-row ml-4">   {/* ml = margin-left (RTL でも左のまま) */}
```

🟡 **SHOULD**: 現時点で RTL サポートが不要でも `ms-` / `ps-` クラスを使う習慣にする

## 8. 型安全な翻訳キー

```ts
// src/types/i18n.d.ts — 翻訳キーの型補完を有効化
import 'i18next';
import ja from '@/locales/ja';

type TranslationResources = typeof ja;

declare module 'i18next' {
  interface CustomTypeOptions {
    defaultNS: 'translation';
    resources: { translation: TranslationResources };
  }
}
```

```tsx
// t() の引数が型補完される状態になる
const { t } = useTranslation();
t('order.status.pending');    // ✅ 補完される
t('order.status.invalid');    // ❌ TypeScript エラー
```

🟡 **SHOULD**: `src/types/i18n.d.ts` を設定して翻訳キーの型補完を有効化する

## 9. テスト実装規約

```tsx
// src/test/utils/i18nWrapper.tsx — テスト用の i18n ラッパー
import { I18nextProvider } from 'react-i18next';
import i18n                from 'i18next';
import { initReactI18next } from 'react-i18next';
import ja                  from '@/locales/ja';

const testI18n = i18n.createInstance();
testI18n.use(initReactI18next).init({
  lng:       'ja',
  resources: { ja: { translation: ja } },
  initImmediate: false,
});

export function I18nWrapper({ children }: { children: React.ReactNode }) {
  return <I18nextProvider i18n={testI18n}>{children}</I18nextProvider>;
}
```

```ts
// テストでのフォーマット関数の検証
describe('formatPrice', () => {
  test.each([
    [1500,  'JPY', '¥1,500'],
    [9.99,  'USD', '$9.99'],
    [1000,  'EUR', '€1,000.00'],
  ])('%d %s → %s', (amount, currency, expected) => {
    expect(formatPrice(amount, currency)).toBe(expected);
  });
});
```
