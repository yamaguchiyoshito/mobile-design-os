## 2. filePicker — 統合ファイル選択

```ts
// src/lib/filePicker.ts
import * as ImagePicker    from 'expo-image-picker';
import * as DocumentPicker from 'expo-document-picker';
import * as FileSystem     from 'expo-file-system';
import { cameraService, mediaLibraryService } from '@/lib/permissions';
import { getFileSize } from '@/lib/fileCache';

// ─── 選択ファイルの共通型 ────────────────────────────────────────────────
export type PickedFile = {
  uri:      string;
  name:     string;
  mimeType: string;
  size:     number;      // bytes
  width?:   number;      // 画像のみ
  height?:  number;      // 画像のみ
};

// ─── バリデーション設定 ──────────────────────────────────────────────────
export type FilePickerConfig = {
  maxSizeBytes?:     number;     // デフォルト: 10MB
  allowedMimeTypes?: string[];   // デフォルト: 制限なし
  // NOTE: allowedExtensions は現バージョンでは未使用。MIME タイプで制御すること。
};

const DEFAULT_CONFIG: Required<Omit<FilePickerConfig, 'allowedExtensions'>> = {
  maxSizeBytes:     10 * 1024 * 1024,  // 10MB
  allowedMimeTypes: [],
};

// ─── バリデーション ──────────────────────────────────────────────────────
function validateFile(
  file: PickedFile,
  config: Required<Omit<FilePickerConfig, 'allowedExtensions'>>,
): string | null {
  if (file.size > config.maxSizeBytes) {
    const maxMB     = Math.round(config.maxSizeBytes / (1024 * 1024));
    const actualMB  = Math.round(file.size / (1024 * 1024));
    return `ファイルサイズは ${maxMB}MB 以下にしてください（現在: ${actualMB}MB）`;
  }
  if (
    config.allowedMimeTypes.length > 0 &&
    !config.allowedMimeTypes.includes(file.mimeType)
  ) {
    return `このファイル形式（${file.mimeType}）は対応していません`;
  }
  return null;
}

// ─── カメラで撮影 ────────────────────────────────────────────────────────
export async function pickFromCamera(
  config?: FilePickerConfig,
): Promise<{ file: PickedFile | null; error: string | null }> {
  const cfg = { ...DEFAULT_CONFIG, ...config };

  try {
    const status = await cameraService.getStatus();
    if (status !== 'granted') {
      const result = await cameraService.request();
      if (result !== 'granted') {
        return { file: null, error: 'カメラへのアクセスが許可されていません' };
      }
    }

    const media = await cameraService.launchCamera();
    if (!media) return { file: null, error: null };  // キャンセル

    const size = await getFileSize(media.uri);

    const file: PickedFile = {
      uri:      media.uri,
      name:     `photo_${Date.now()}.jpg`,
      mimeType: media.mimeType,
      size,
      width:    media.width,
      height:   media.height,
    };

    const error = validateFile(file, cfg);
    return { file: error ? null : file, error };
  } catch (err) {
    console.error('[filePicker] pickFromCamera error:', err);
    return { file: null, error: 'カメラの起動中にエラーが発生しました' };
  }
}

// ─── フォトライブラリから選択 ────────────────────────────────────────────
// NOTE: launchImageLibrary は cameraService ではなく mediaLibraryService が担当する
export async function pickFromLibrary(
  config?: FilePickerConfig & { multiple?: boolean },
): Promise<{ files: PickedFile[]; errors: string[] }> {
  const cfg      = { ...DEFAULT_CONFIG, ...config };
  const multiple = config?.multiple ?? false;

  try {
    const status = await mediaLibraryService.getStatus();
    if (status === 'denied') {
      return { files: [], errors: ['フォトライブラリへのアクセスが許可されていません'] };
    }
    if (status === 'undetermined') {
      const result = await mediaLibraryService.request();
      if (result === 'denied') {
        return { files: [], errors: ['フォトライブラリへのアクセスが許可されていません'] };
      }
    }

    // ✅ mediaLibraryService.launchImageLibrary を使用（cameraService ではない）
    const assets = await mediaLibraryService.launchImageLibrary({ allowsMultipleSelection: multiple });
    if (!assets) return { files: [], errors: [] };  // キャンセル

    const files: PickedFile[] = [];
    const errors: string[]    = [];

    for (const asset of assets) {
      const size = await getFileSize(asset.uri);
      const file: PickedFile = {
        uri:      asset.uri,
        name:     asset.uri.split('/').pop() ?? `image_${Date.now()}.jpg`,
        mimeType: asset.mimeType,
        size,
        width:    asset.width,
        height:   asset.height,
      };
      const err = validateFile(file, cfg);
      if (err) errors.push(err);
      else files.push(file);
    }

    return { files, errors };
  } catch (err) {
    console.error('[filePicker] pickFromLibrary error:', err);
    return { files: [], errors: ['フォトライブラリの起動中にエラーが発生しました'] };
  }
}

// ─── ドキュメント選択（PDF / Excel / Word 等）──────────────────────────
// NOTE: copyToCacheDirectory: true — iCloud Drive 等サンドボックス外の URI は
//       アプリが直接アクセスできないため、アプリキャッシュ領域にコピーしてから返す。
//       コピーされたファイルはアップロード完了後に deleteFromCache() で削除すること。
export async function pickDocument(
  config?: FilePickerConfig & { multiple?: boolean },
): Promise<{ files: PickedFile[]; errors: string[] }> {
  const cfg = { ...DEFAULT_CONFIG, ...config };

  // ✅ type は string[] | string を正しく分岐して渡す
  const mimeTypes: string | string[] =
    cfg.allowedMimeTypes.length > 0 ? cfg.allowedMimeTypes : '*/*';

  const result = await DocumentPicker.getDocumentAsync({
    type:                 mimeTypes,
    multiple:             config?.multiple ?? false,
    copyToCacheDirectory: true,
  });

  if (result.canceled) return { files: [], errors: [] };

  const files: PickedFile[] = [];
  const errors: string[]    = [];

  for (const asset of result.assets) {
    const file: PickedFile = {
      uri:      asset.uri,
      name:     asset.name,
      mimeType: asset.mimeType ?? 'application/octet-stream',
      size:     asset.size ?? 0,
    };
    const err = validateFile(file, cfg);
    if (err) errors.push(`${asset.name}: ${err}`);
    else files.push(file);
  }

  return { files, errors };
}
```

## 3. 一時ファイル・キャッシュ管理

```ts
// src/lib/fileCache.ts
import * as FileSystem from 'expo-file-system';

const CACHE_DIR = `${FileSystem.cacheDirectory}app-uploads/`;

// ─── キャッシュディレクトリの初期化 ──────────────────────────────────────
export async function ensureCacheDir(): Promise<void> {
  const info = await FileSystem.getInfoAsync(CACHE_DIR);
  if (!info.exists) {
    await FileSystem.makeDirectoryAsync(CACHE_DIR, { intermediates: true });
  }
}

// ─── ファイルサイズ取得（共通ユーティリティ）────────────────────────────
// filePicker / imageUtils から共通利用する。FileSystem.FileInfo の size は
// { size: true } オプション指定時のみ付与されるため、ここで型安全に取得する。
export async function getFileSize(uri: string): Promise<number> {
  const info = await FileSystem.getInfoAsync(uri, { size: true });
  return (info as FileSystem.FileInfo & { size?: number }).size ?? 0;
}

// ─── ファイルをキャッシュにコピー ────────────────────────────────────────
// 元の URI へのアクセス権が消えた後も使えるようにアプリキャッシュ領域へコピーする。
export async function copyToCache(uri: string, filename: string): Promise<string> {
  await ensureCacheDir();
  const dest = `${CACHE_DIR}${filename}`;
  await FileSystem.copyAsync({ from: uri, to: dest });
  return dest;
}

// ─── アップロード済みファイルのキャッシュを削除 ──────────────────────────
export async function deleteFromCache(uri: string): Promise<void> {
  const info = await FileSystem.getInfoAsync(uri);
  if (info.exists) {
    await FileSystem.deleteAsync(uri, { idempotent: true });
  }
}

// ─── キャッシュを全クリア（ログアウト時等）──────────────────────────────
export async function clearCache(): Promise<void> {
  const info = await FileSystem.getInfoAsync(CACHE_DIR);
  if (info.exists) {
    await FileSystem.deleteAsync(CACHE_DIR, { idempotent: true });
  }
}
```

🟡 **SHOULD**: ログアウト時に `fileCache.clearCache()` を呼ぶ  
🟡 **SHOULD**: アップロード完了後は `deleteFromCache()` で一時ファイルを削除する

## 4. アップロード付き選択フロー（Feature パターン）

```tsx
// features/documents/components/DocumentUploadButton.tsx
import { pickDocument }   from '@/lib/filePicker';
import { uploadFile }     from '@/lib/fileUpload';
import { copyToCache, deleteFromCache } from '@/lib/fileCache';

const ALLOWED_TYPES  = ['application/pdf', 'image/jpeg', 'image/png'];
const MAX_SIZE_BYTES = 20 * 1024 * 1024;  // 20MB（PDF対応のため拡張）

export function DocumentUploadButton({ onUploaded }: { onUploaded: (url: string) => void }) {
  const { t }                       = useTranslation();
  const [progress, setProgress]     = useState<number | null>(null);
  const [error,    setError]        = useState<string | null>(null);

  const handlePick = async () => {
    setError(null);

    const { files, errors } = await pickDocument({
      allowedMimeTypes: ALLOWED_TYPES,
      maxSizeBytes:     MAX_SIZE_BYTES,
    });

    if (errors.length > 0) {
      setError(errors[0]);
      return;
    }
    // バリデーション通過ファイルなし（キャンセル含む）
    if (files.length === 0) return;

    const file   = files[0];
    const cached = await copyToCache(file.uri, file.name);

    try {
      const result = await uploadFile(cached, '/documents', {
        onProgress: setProgress,
      });

      onUploaded(result.url);
      notify.success(t('document.uploadSuccess'));
    } catch (err) {
      notify.error(t('document.uploadError'));
    } finally {
      // 成功・失敗どちらでも一時ファイルを削除する
      await deleteFromCache(cached);
      setProgress(null);
    }
  };

  return (
    <View>
      <Button
        label={t('document.selectFile')}
        variant="secondary"
        onPress={handlePick}
        disabled={progress !== null}
        loading={progress !== null}
      />

      {progress !== null && (
        <View
          accessibilityLiveRegion="polite"
          accessibilityLabel={`アップロード中 ${Math.round(progress * 100)}%`}
        >
          <ProgressBar value={progress} />
          <Text className="text-body-sm text-content-secondary">
            {Math.round(progress * 100)}%
          </Text>
        </View>
      )}

      {error && <FieldError message={error} />}
    </View>
  );
}
```

## 5. 画像ユーティリティ（リサイズ・EXIF除去）

`resizeImage` は UI コンポーネントではなくユーティリティ関数のため、専用モジュールに分離する。

```ts
// src/lib/imageUtils.ts
import * as ImageManipulator from 'expo-image-manipulator';

type ResizeOptions = {
  maxWidth?:  number;   // デフォルト: 1200px
  maxHeight?: number;   // デフォルト: 1200px
  quality?:   number;   // デフォルト: 0.8（0〜1）
};

// ─── リサイズ・圧縮・EXIF除去 ────────────────────────────────────────────
// expo-image-manipulator で再エンコードすることで EXIF（位置情報等）が除去される。
// カメラ撮影画像は cameraService の exif: false で除去済みだが、
// フォトライブラリ経由の画像は EXIF を含む場合があるため、送信前に必ずこの関数を通すこと。
export async function resizeImage(uri: string, options: ResizeOptions = {}): Promise<string> {
  const { maxWidth = 1200, maxHeight = 1200, quality = 0.8 } = options;

  const result = await ImageManipulator.manipulateAsync(
    uri,
    [{ resize: { width: maxWidth, height: maxHeight } }],
    { compress: quality, format: ImageManipulator.SaveFormat.JPEG },
  );
  return result.uri;
}
```

🟡 **SHOULD**: サーバー送信前に `resizeImage` で最大 1200px・品質 0.8 に圧縮する  
🔴 **MUST**: フォトライブラリ経由の画像は送信前に `resizeImage` を通し EXIF を除去する
